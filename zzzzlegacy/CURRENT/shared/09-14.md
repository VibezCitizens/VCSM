create or replace function reviews.validate_target_actor_is_active_vport(p_target_actor_id uuid)
returns void
language plpgsql
security definer
set search_path = reviews, vc, public
set row_security = off
as $$
begin
  if not exists (
    select 1
    from vc.actors a
    join vc.vports v
      on v.id = a.vport_id
    where a.id = p_target_actor_id
      and a.kind = 'vport'
      and coalesce(a.is_void, false) = false
      and coalesce(v.is_active, false) = true
  ) then
    raise exception 'target_actor_id must reference an active vport actor';
  end if;
end;
$$;

returns text

as $$
  select v.vport_type
  from vc.actors a
  join vc.vports v
    on v.id = a.vport_id
  where a.id = p_target_actor_id
  limit 1
$$;

create or replace function reviews.hydrate_review_snapshots()
returns trigger
language plpgsql
security definer
set search_path = reviews, vc, public, identity
set row_security = off
as $$
declare
  v_author_kind text;
  v_author_profile_id uuid;
  v_author_vport_id uuid;

  v_target_name text;
  v_target_slug text;
  v_target_avatar text;

  v_author_display text;
  v_author_username text;
  v_author_avatar text;
begin
  perform reviews.validate_target_actor_is_active_vport(new.target_actor_id);

  new.target_kind := 'vport';
  new.target_subtype := reviews.resolve_target_subtype(new.target_actor_id);

  select
    v.name,
    v.slug,
    nullif(v.avatar_url, '')
  into
    v_target_name,
    v_target_slug,
    v_target_avatar
  from vc.actors a
  join vc.vports v
    on v.id = a.vport_id
  where a.id = new.target_actor_id;

  new.target_display_name_snapshot := v_target_name;
  new.target_username_snapshot := v_target_slug;
  new.target_avatar_url_snapshot := v_target_avatar;

  select
    a.kind,
    a.profile_id,
    a.vport_id
  into
    v_author_kind,
    v_author_profile_id,
    v_author_vport_id
  from vc.actors a
  where a.id = new.author_actor_id
    and coalesce(a.is_void, false) = false;

  if v_author_kind is null then
    raise exception 'author_actor_id is invalid or void';
  end if;

  if v_author_kind = 'user' then
    select
      coalesce(nullif(trim(ad.display_name), ''), nullif(trim(p.display_name), ''), 'Anonymous'),
      coalesce(nullif(trim(ad.username), ''), nullif(trim(p.username), '')),
      coalesce(nullif(ad.avatar_url, ''), nullif(p.photo_url, ''))
    into
      v_author_display,
      v_author_username,
      v_author_avatar
    from public.profiles p
    left join identity.actor_directory ad
      on ad.actor_id = new.author_actor_id
    where p.id = v_author_profile_id;
  else
    select
      coalesce(nullif(trim(v.name), ''), 'Anonymous'),
      nullif(trim(v.slug), ''),
      nullif(v.avatar_url, '')
    into
      v_author_display,
      v_author_username,
      v_author_avatar
    from vc.vports v
    where v.id = v_author_vport_id;
  end if;

  new.author_display_name_snapshot := coalesce(v_author_display, 'Anonymous');
  new.author_username_snapshot := v_author_username;
  new.author_avatar_url_snapshot := v_author_avatar;

  if tg_op = 'INSERT' then
    new.review_activity_at := coalesce(new.review_activity_at, now());
  end if;

  return new;
end;
$$;

create trigger trg_reviews_hydrate_snapshots
before insert or update of target_actor_id, author_actor_id
on reviews.reviews
for each row
execute function reviews.hydrate_review_snapshots();

-- =========================================================
-- NEUTRAL REVIEW CREATE COOLDOWN
-- current rule:
-- author may CREATE at most one new neutral review every 24 hours
-- updates to an existing card are allowed anytime
-- =========================================================

create or replace function reviews.enforce_neutral_review_create_cooldown()
returns trigger
language plpgsql
security definer
set search_path = reviews, public
set row_security = off
as $$
begin
  if new.review_mode <> 'neutral' then
    return new;
  end if;

  if exists (
    select 1
    from reviews.reviews r
    where r.author_actor_id = new.author_actor_id
      and r.review_mode = 'neutral'
      and r.is_deleted = false
      and r.created_at > now() - interval '24 hours'
  ) then
    raise exception 'neutral review create cooldown is 24 hours per author';
  end if;

  return new;
end;
$$;

create trigger trg_reviews_neutral_create_cooldown
before insert on reviews.reviews
for each row
execute function reviews.enforce_neutral_review_create_cooldown();

-- =========================================================

create or replace function reviews.hydrate_dimension_rating_snapshots()
returns trigger
language plpgsql
security definer
set search_path = reviews, public
set row_security = off
as $$
declare
  v_label text;
  v_weight numeric(10,4);
  v_review_target_subtype text;
  v_dimension_target_subtype text;
  v_dimension_target_kind text;
begin
  select r.target_subtype
  into v_review_target_subtype
  from reviews.reviews r
  where r.id = new.review_id;

  select
    d.label,
    d.weight,
    d.target_subtype,
    d.target_kind
  into
    v_label,
    v_weight,
    v_dimension_target_subtype,
    v_dimension_target_kind
  from reviews.review_dimensions d
  where d.id = new.dimension_id
    and d.is_active = true;

  if v_label is null then
    raise exception 'dimension_id is invalid or inactive';
  end if;

  if v_dimension_target_kind <> 'vport' then
    raise exception 'dimension target_kind mismatch';
  end if;

  if v_review_target_subtype is distinct from v_dimension_target_subtype then
    raise exception 'dimension target_subtype mismatch for review';
  end if;

  new.label_snapshot := coalesce(new.label_snapshot, v_label);
  new.weight_snapshot := coalesce(new.weight_snapshot, v_weight, 1.0);

  return new;
end;
$$;

create trigger trg_review_dimension_ratings_hydrate_snapshots
before insert on reviews.review_dimension_ratings
for each row
execute function reviews.hydrate_dimension_rating_snapshots();

-- =========================================================
-- OVERALL RATING RECALC
-- weighted average across current dimension ratings
-- =========================================================

create or replace function reviews.recalc_review_overall_rating(p_review_id uuid)
returns void
language plpgsql
security definer
set search_path = reviews, public
set row_security = off
as $$
begin
  update reviews.reviews r
  set
    overall_rating = sub.weighted_avg,
    review_activity_at = now(),
    updated_at = now()
  from (
    select
      case
        when coalesce(sum(rr.weight_snapshot), 0) = 0 then null
        else round(sum((rr.rating::numeric) * rr.weight_snapshot) / sum(rr.weight_snapshot), 2)
      end as weighted_avg
    from reviews.review_dimension_ratings rr
    where rr.review_id = p_review_id
  ) sub
  where r.id = p_review_id;
end;
$$;

create or replace function reviews.trg_recalc_review_overall_rating()
returns trigger
language plpgsql
security definer
set search_path = reviews, public
set row_security = off
as $$
begin
  perform reviews.recalc_review_overall_rating(coalesce(new.review_id, old.review_id));
  return coalesce(new, old);
end;
$$;

create trigger trg_review_dimension_ratings_recalc
after insert or update or delete
on reviews.review_dimension_ratings
for each row
execute function reviews.trg_recalc_review_overall_rating();

-- =========================================================
-- REVIEW REVISION HISTORY
-- =========================================================

create or replace function reviews.write_review_revision()
returns trigger
language plpgsql
security definer
set search_path = reviews, public
set row_security = off
as $$
declare
  v_next_revision integer;
begin
  select coalesce(max(rr.revision_no), 0) + 1
  into v_next_revision
  from reviews.review_revisions rr
  where rr.review_id = coalesce(new.id, old.id);

  if tg_op = 'INSERT' then
    insert into reviews.review_revisions (
      review_id,
      revision_no,
      change_kind,
      body,
      overall_rating,
      review_activity_at,
      snapshot_json
    )
    values (
      new.id,
      v_next_revision,
      'created',
      new.body,
      new.overall_rating,
      new.review_activity_at,
      jsonb_build_object(
        'author_display_name_snapshot', new.author_display_name_snapshot,
        'author_username_snapshot', new.author_username_snapshot,
        'author_avatar_url_snapshot', new.author_avatar_url_snapshot,
        'target_display_name_snapshot', new.target_display_name_snapshot,
        'target_username_snapshot', new.target_username_snapshot,
        'target_avatar_url_snapshot', new.target_avatar_url_snapshot,
        'review_mode', new.review_mode,
        'verification_status', new.verification_status
      )
    );
    return new;
  end if;

  if tg_op = 'UPDATE' then
    if
      new.body is distinct from old.body
      or new.overall_rating is distinct from old.overall_rating
      or new.review_activity_at is distinct from old.review_activity_at
      or new.is_deleted is distinct from old.is_deleted
    then
      insert into reviews.review_revisions (
        review_id,
        revision_no,
        change_kind,
        body,
        overall_rating,
        review_activity_at,
        snapshot_json
      )
      values (
        new.id,
        v_next_revision,
        case when new.is_deleted = true and old.is_deleted = false then 'deleted' else 'updated' end,
        new.body,
        new.overall_rating,
        new.review_activity_at,
        jsonb_build_object(
          'author_display_name_snapshot', new.author_display_name_snapshot,
          'author_username_snapshot', new.author_username_snapshot,
          'author_avatar_url_snapshot', new.author_avatar_url_snapshot,
          'target_display_name_snapshot', new.target_display_name_snapshot,
          'target_username_snapshot', new.target_username_snapshot,
          'target_avatar_url_snapshot', new.target_avatar_url_snapshot,
          'review_mode', new.review_mode,
          'verification_status', new.verification_status
        )
      );
    end if;

    return new;
  end if;

  return coalesce(new, old);
end;
$$;

-- =========================================================
-- OFFICIAL / CURRENT OVERALL
-- today includes both neutral + transactional if active and not deleted
-- later you can weight transactional more heavily by changing this function
-- =========================================================

create or replace function reviews.get_target_overall_stats(p_target_actor_id uuid)
returns table (
  target_actor_id uuid,
  review_count bigint,
  neutral_review_count bigint,
  transactional_review_count bigint,
  overall_avg numeric,
  overall_p50 numeric,
  overall_p90 numeric
)
language sql
stable
security definer
set search_path = reviews, public
set row_security = off
as $$
  with active_reviews as (
    select r.*
    from reviews.reviews r
    where r.target_actor_id = p_target_actor_id
      and r.is_deleted = false
      and r.active_card = true
      and r.overall_rating is not null
  )
  select
    p_target_actor_id,
    count(*)::bigint,
    count(*) filter (where review_mode = 'neutral')::bigint,
    count(*) filter (where review_mode = 'transactional')::bigint,
    round(avg(overall_rating)::numeric, 4),
    round(percentile_cont(0.50) within group (order by overall_rating)::numeric, 4),
    round(percentile_cont(0.90) within group (order by overall_rating)::numeric, 4)
  from active_reviews
$$;

grant execute on function reviews.get_target_overall_stats(uuid) to authenticated;

-- =========================================================
-- OPTIONAL UPSERT FUNCTION
-- current behavior:
-- if active neutral review exists for same author->target, update same card
-- otherwise insert new card
-- caller can then upsert ratings separately
-- =========================================================

create or replace function reviews.upsert_neutral_review(
  p_target_actor_id uuid,
  p_author_actor_id uuid,
  p_body text
)
returns uuid
language plpgsql
security definer
set search_path = reviews, vc, public
set row_security = off
as $$
declare
  v_review_id uuid;
begin
  perform reviews.require_authenticated();

  if not vc.is_actor_owner(p_author_actor_id) then
    raise exception 'not authorized to write review as this actor';
  end if;

  perform reviews.validate_target_actor_is_active_vport(p_target_actor_id);

  select r.id
  into v_review_id
  from reviews.reviews r
  where r.target_actor_id = p_target_actor_id
    and r.author_actor_id = p_author_actor_id
    and r.review_mode = 'neutral'
    and r.active_card = true
    and r.is_deleted = false
  limit 1;

  if v_review_id is null then
    insert into reviews.reviews (
      target_actor_id,
      author_actor_id,
      review_mode,
      verification_status,
      body
    )
    values (
      p_target_actor_id,
      p_author_actor_id,
      'neutral',
      'unverified',
      coalesce(p_body, '')
    )
    returning id into v_review_id;
  else
    update reviews.reviews r
    set
      body = coalesce(p_body, ''),
      review_activity_at = now()
    where r.id = v_review_id;
  end if;

  return v_review_id;
end;
$$;

grant execute on function reviews.upsert_neutral_review(uuid, uuid, text) to authenticated;

### Why This Approach
- Matches identity/chat engine patterns exactly (config DI, DAL/model/controller layers, adapters-only public API)
- Engine owns `reviews.*` schema exclusively — no cross-schema queries
- Actor ownership injected via `isActorOwner` (app provides implementation)
- Uses existing DB RPC functions (`upsert_neutral_review`, `get_review_author_card`, `get_target_overall_stats`)
- Snapshot hydration handled by DB triggers — engine trusts trigger-enriched data

### Key Design Decisions
- RPC-first for upsert (leverages DB-level cooldown enforcement and snapshot hydration)
- Cursor-based pagination via `review_activity_at` (matches existing app pattern)
- Author cards fetched via SECURITY DEFINER RPC (bypasses RLS for private actors)
- No React hooks in engine — app responsibility to wire UI lifecycle
